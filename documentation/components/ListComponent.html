import { Component, Input, OnInit } from '@angular/core';
import { select, Store } from '@libs/midgard-angular/src/lib/modules/store/store';

@Component({
  selector: 'mg-crud-list',
  templateUrl: './list.component.html',
  styleUrls: ['./list.component.css']
})
export class ListComponent implements OnInit {

  public rows = [];
  @Input() tableOptions; // options for the table component
  @Input() loadAction; // redux action to load data
  @Input() reducer; // redux reducer to get data from
  @Input() model; // key of which value will be returned from the reducer

  constructor(
    private store: Store<any>, // type {any} beacuse the state of the app is not fixed and can be changed depending on the modules
  ) { }

  ngOnInit() {
    this.store.dispatch({
      type: this.loadAction,
    });
    this.store.observable.pipe(
      select(this.reducer, this.model)
    ).subscribe( (data: any[]) => {
      this.rows = data;
    });
  }

}
